function CopyBtn({ text }) {
  const [done, setDone] = useState(false);
  const copy = () => {
    navigator.clipboard.writeText(text).then(() => { setDone(true); setTimeout(() => setDone(false), 2000); });
  };
  return (
    <button onClick={copy} style={{
      padding:"7px 16px", borderRadius:8, cursor:"pointer",
      border:`1px solid ${done ? T.green+"60" : T.border}`,
      background: done ? `${T.green}12` : "transparent",
      color: done ? T.green : T.dim,
      fontFamily:T.mono, fontSize:11, transition:"all 0.2s", flexShrink:0,
    }}>
      {done ? "✓ تم النسخ" : "نسخ ↗"}
    </button>
  );
}
